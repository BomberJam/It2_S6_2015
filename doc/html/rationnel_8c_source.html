159 void print_rationnel(Rationnel* rat)
160 {
161  if (rat == NULL)
162  {
163  printf("∅");
164  return;
165  }
166 
167  switch(get_etiquette(rat))
168  {
169  case EPSILON:
170  printf("ε");
171  break;
172 
173  case LETTRE:
174  printf("%c", get_lettre(rat));
175  break;
176 
177  case UNION:
178  printf("(");
179  print_rationnel(fils_gauche(rat));
180  printf(" + ");
181  print_rationnel(fils_droit(rat));
182  printf(")");
183  break;
184 
185  case CONCAT:
186  printf("[");
187  print_rationnel(fils_gauche(rat));
188  printf(" . ");
189  print_rationnel(fils_droit(rat));
190  printf("]");
191  break;
192 
193  case STAR:
194  printf("{");
195  print_rationnel(fils(rat));
196  printf("}*");
197  break;
198 
199  default:
200  assert(false);
201  break;
202  }
203 }

205 Rationnel *expression_to_rationnel(const char *expr)
206 {
207  Rationnel *rat;
208  yyscan_t scanner;
209  YY_BUFFER_STATE state;
210 
211  // Initialisation du scanner
212  if (yylex_init(&scanner))
213  return NULL;
214 
215  state = yy_scan_string(expr, scanner);
216 
217  // Test si parsing ok.
218  if (yyparse(&rat, scanner))
219  return NULL;
220 
221  // Libération mémoire
222  yy_delete_buffer(state, scanner);
223 
224  yylex_destroy(scanner);
225 
226  return rat;
227 }

235 int rationnel_to_dot_aux(Rationnel *rat, FILE *output, int pere, int noeud_courant)
236 {
237  int saved_pere = noeud_courant;
238 
239  if (pere >= 1)
240  fprintf(output, "\tnode%d -> node%d;\n", pere, noeud_courant);
241  else
242  fprintf(output, "digraph G{\n");
243 
244  switch(get_etiquette(rat))
245  {
246  case LETTRE:
247  fprintf(output, "\tnode%d [label = \"%c-%d\"];\n", noeud_courant, get_lettre(rat), rat->position_min);
248  noeud_courant++;
249  break;
250 
251  case EPSILON:
252  fprintf(output, "\tnode%d [label = \"ε-%d\"];\n", noeud_courant, rat->position_min);
253  noeud_courant++;
254  break;
255 
256  case UNION:
257  fprintf(output, "\tnode%d [label = \"+ (%d/%d)\"];\n", noeud_courant, rat->position_min, rat->position_max);
258  noeud_courant = rationnel_to_dot_aux(fils_gauche(rat), output, noeud_courant, noeud_courant+1);
259  noeud_courant = rationnel_to_dot_aux(fils_droit(rat), output, saved_pere, noeud_courant+1);
260  break;
261 
262  case CONCAT:
263  fprintf(output, "\tnode%d [label = \". (%d/%d)\"];\n", noeud_courant, rat->position_min, rat->position_max);
264  noeud_courant = rationnel_to_dot_aux(fils_gauche(rat), output, noeud_courant, noeud_courant+1);
265  noeud_courant = rationnel_to_dot_aux(fils_droit(rat), output, saved_pere, noeud_courant+1);
266  break;
267 
268  case STAR:
269  fprintf(output, "\tnode%d [label = \"* (%d/%d)\"];\n", noeud_courant, rat->position_min, rat->position_max);
270  noeud_courant = rationnel_to_dot_aux(fils(rat), output, noeud_courant, noeud_courant+1);
271  break;
272 
273  default:
274  assert(false);
275  break;
276  }
277  if (pere < 0)
278  fprintf(output, "}\n");
279  return noeud_courant;
280 }
